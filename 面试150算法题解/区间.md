## 1、汇总区间（简单）
## 没啥好说，读题都快点
```C++

class Solution {
public:
    vector<string> summaryRanges(vector<int>& nums) {
        int numsSize = nums.size();
        string jt = "->";
        int i = 0;
        vector<string> result;
        while(i < numsSize){
            string first = to_string(nums[i]);
            while((i+1) < numsSize && nums[i+1] == nums[i]+1){
                i++;
            }
            string last = to_string(nums[i]);
            string subInterval;
            if(first != last)
                subInterval = first+jt+last;
            else
                subInterval = first;
            result.push_back(subInterval);
            i++;
        }
        return result;
    }
};
```

## 2、合并区间（中等）

### 个人思路：

自己能相差不大，记住先排序就好。

```C++
class Solution {
public:
    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        int intervalsSize = intervals.size();
        if (intervalsSize == 0)
            return {};
        sort(intervals.begin(), intervals.end());
        vector<vector<int>> result;
        
        for (int i = 0; i < intervalsSize; i++) {
            int L = intervals[i][0];
            int R = intervals[i][1];
            
            if (result.size() == 0 || result.back()[1] < L) {
                result.push_back({ L, R });
            }
            else {
                result.back()[1] = max(result.back()[1], R);
            }
        }
        return result;
    }

};
```
